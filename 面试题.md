function convert(money) {
	return money.toString().replace(/\B(?=([0-9]{3})+$)/g,',');
}

function convert(money) {
	let str = money.toString();
  let res = [];
  let start = str.length%3;
  if(start !== 0){
    res.push(str.substr(0, start));
  }
  while(start < str.length) {
    res.push(str.substr(start, 3));
    start += 3;
  }
  return res.join(',');
}
